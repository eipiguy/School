\begin{problem}
Find the principal argument, $\arg (z)$, when
\begin{enumerate}[(a)]
\item $z = \frac{i}{-2-2i}$
\item $z = (\sqrt{3}-i)^{6}$
\end{enumerate}
\end{problem}

\begin{solution}
\begin{enumerate}[(a)]
\item  Since 
\[ z = \frac{i}{-2-2i} = \frac{i(-2+2i)}{4+4} = \frac{-(1+i)}{4} \]
Then the point has principal argument $\arg(z)=-\pi/3$
\item Here we use Pascal's formula to see$z = (\sqrt{3}-i)^{6}$
\end{enumerate}
\end{solution}

\begin{problem}

\begin{problem}
Sketch the following sets and determine which are domains:
\begin{enumerate}[(a)]
\item $\text{Im}(z)>1$
\item $\text{Im}(z)=1$
\item $0 \le \arg(z) \le \pi/4$, $(z\ne0)$
\item $|z-4| \ge |z|$
\end{enumerate}
\end{problem}

\begin{problem}

\item $\text{Re}(z^{2}) > 0$
\end{enumerate}
\end{problem}
